DATABASE STRUCTURE:
================================

TABLE 1: users
Columns:
- id (INT, Primary Key, Auto Increment)
- first_name (VARCHAR 100)
- middle_name (VARCHAR 100) - Optional
- last_name (VARCHAR 100)
- email (VARCHAR 150) - UNIQUE
- student_id (VARCHAR 50) - UNIQUE
- department (VARCHAR 100)
- password (VARCHAR 255) - Hashed
- has_voted (TINYINT 0 or 1) - Default: 0
- registration_date (TIMESTAMP)

Indexes:
- idx_student_id on student_id
- idx_email on email

TABLE 2: votes
Columns:
- vote_id (INT, Primary Key, Auto Increment)
- student_id (VARCHAR 50) - Foreign Key
- president (VARCHAR 100)
- vice_president (VARCHAR 100)
- secretary (VARCHAR 100)
- timestamp (TIMESTAMP)

Indexes:
- idx_vote_student on student_id
- idx_vote_timestamp on timestamp

DATABASE QUERIES FOR TESTING:
================================

View all students:
SELECT * FROM users;

View specific student:
SELECT * FROM users WHERE student_id = 'CS001';

View who has voted:
SELECT first_name, last_name, student_id, has_voted FROM users;

View all votes cast:
SELECT * FROM votes;

View votes for President:
SELECT president, COUNT(*) as votes FROM votes GROUP BY president;

View votes for Vice President:
SELECT vice_president, COUNT(*) as votes FROM votes GROUP BY vice_president;

View votes for Secretary:
SELECT secretary, COUNT(*) as votes FROM votes GROUP BY secretary;

View vote count per user:
SELECT student_id, COUNT(*) as vote_count FROM votes GROUP BY student_id;

Reset a user (allow to vote again):
UPDATE users SET has_voted = 0 WHERE student_id = 'CS001';

Delete all votes:
DELETE FROM votes;

Delete all users:
DELETE FROM users;
